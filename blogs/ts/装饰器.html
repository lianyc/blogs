    装饰器

    12/27/2023 typescript装饰器

    • 装饰器可以分离关注点：在定义时最清楚要定义的属性或方法的情况

    • 作用：为属性、方法、类、参数提供元数据（metadata:描述数据的数据）信息

    • TS中使用装饰器，需要开启*"experimentalDecorators"*: true

    # 类装饰器

    装饰器的本质是一个函数，该函数接收一个参数，表示类本身（构造函数本身）

    使用装饰器：@装饰器名

    TS中，如何 约束一个变量为类：

    • Function
    • new(参数) => object

    装饰器函数的运行时间：在类定义后直接运行

    类装饰器可以具有的返回值：

    • void
    • 返回一个新的类：会将新的类替换掉装饰目标
    import { log } from "console";

function test(target: new()=>object) {
  log(target);
  return class B {
    constructor() {
      log("B");
    }
  }
}
@test
class A {
  constructor() {
    log("A");
  }
}

const a = new A(); // B

    function test(target: new(...args: any[])=>object) {}
@test
class A {
  constructor(public p:string) {}
}

    import { log } from "console"

function test(str:string){
  log(str)
  return function(target: new(...args: any[])=>object) {}
}

@test("hello world")
class A {
  constructor(public p:string) {}
}

    多个装饰器运行的情况:

    • 装饰器函数后加入先调用

    • 返回装饰器的函数从上往下运行

    import { log } from "console"
type constructor = new(...args: any[])=>object
function d1(str:string){
  log(str)
  return function(target:constructor) {
    log('d1')
  }
}

function d2(str:string){
  log(str)
  return function(target:constructor) {
    log('d2')
  }
}

@d1("hello world 111")
@d2("hello world 222")
class A {
  constructor(public p:string) {}
}
// hello world 111
// hello world 222
// d2
// d1

    # 成员装饰器

    • 属性: 属性装饰器也是一个函数，该函数需要两个参数

      • 第一个参数：如果是静态属性，则为类本身，如果是实例属性，则为类的原型

      • 第二个参数：固定为字符串，表示属性名

        import { log } from "console"
type constructor = new(...args: any[])=>object
function d(target:any, key:string){
  if(!target.__props) {
    target.__props = []
  }
  target.__props.push(key)
}

class A {
  @d
  static prop1: number;
  @d
  prop2: string;
}

log((A.prototype as any).__props)
const a = new A()
log((a as any).__props)

    • 方法:方法装饰器也是一个函数，该函数需要三个参数

      • 第一个参数：如果是静态方法，则为类本身，如果是实例方法，则为类的原型

      • 第二个参数：固定为字符串，表示方法名

      • 第三个参数：属性描述对象（可选）

        import { log, warn } from "console"
function d(str: string){
  return function(target:any, key:string, descriptor:PropertyDescriptor){
    log(str, target, key, descriptor)
    descriptor.enumerable = true
  }
}

function useLess(target:any, key:string, descriptor:PropertyDescriptor){
  descriptor.value = function(){
    warn(key + '方法已过期')
  }
}

class A {
  @d('f1')
  @useLess
  f1(){
    log('f1 uuuu')
  }

  @d('f2')
  static f2(){}
}

const a = new A()
for (const key in a) {
  log(key)
}
a.f1()

    # 类和属性的描述装饰器

    import { log } from "console";

type constructor = new (...args: any[]) => object;

function classDescriptor(describe: string) {
  return function (target: constructor) {
    // 保存到该类的原型中
    target.prototype.$classDescribe = describe;
  }
}

function propDescriptor(describe: string) {
  return function (target: any, propertyKey: string) {
    if(!target.$props) {
      target.$props = []
    }
    target.$props.push({
      describe,
      propertyKey
    })
  }
}

function printObj(obj: any) {
  if(obj.$classDescribe) {
    log(obj.$classDescribe)
  } else{
    log(obj.__proto__.constructor.name)
  }
  if(!obj.$props) {
    obj.$props = []
  }
  for (const key in obj) {
    const describe = obj.$props.find((item: { propertyKey: string; }) => item.propertyKey === key)?.describe;
    if(describe && obj.hasOwnProperty(key)){
      log('\t' + describe + ':' + obj[key])
    } else if (obj.hasOwnProperty(key)) {
      log('\t'+ key + ':' + obj[key])
    }
  }
}

@classDescriptor('用户')
class User{
  @propDescriptor('用户名')
  userName: string="lian";
  @propDescriptor('密码')
  password: string="123qwe";
}

const u = new User();
printObj(u);

    # reflect-metadata库 (opens new window)

    专门保存元数据

    npm i reflect-metadata

    import "reflect-metadata";
import { log } from 'console';

@Reflect.metadata("className", "类A")
@Reflect.metadata("classType", "类的类型")
class A {
  @Reflect.metadata("propName", "属性prop")
  @Reflect.metadata("propType", "属性type")
  prop: string;
}

const a = new A();

log(Reflect.getMetadata("className", A))
log(Reflect.getMetadata("propName", a, 'prop'))
log(Reflect.getMetadataKeys(A))
log(Reflect.getMetadataKeys(a, 'prop'))

    import { log } from "console";
import 'reflect-metadata'

const key = Symbol.for('describe')

function descriptor(describe: string) {
  return Reflect.metadata(key, describe)
}

function printObj(obj: any) {
  const constructor = Object.getPrototypeOf(obj).constructor
  if(Reflect.hasMetadata(key, constructor)) {
    log(Reflect.getMetadata(key, constructor))
  } else{
    log(constructor.name)
  }
  for (const k in obj) {
    if(Reflect.hasMetadata(key, obj, k)){
      log('\t' + Reflect.getMetadata(key, obj, k) + ':' + obj[k])
    } else {
      log('\t'+ k + ':' + obj[k])
    }
  }
}

@descriptor('用户')
class User{
  @descriptor('用户名')
  userName: string="lian";
  @descriptor('密码')
  password: string="123qwe";
}

const u = new User();
printObj(u);

    # class-validator (opens new window)

    npm i class-validator

    import { log } from "console";
import 'reflect-metadata'
import { Length, IsNotEmpty, validate, Min, MinLength, Max, IsPhoneNumber, IsMobilePhone, Matches } from "class-validator";


class User{
  @IsNotEmpty({message: '用户名不能为空'})
  @Length(4,10,{message: '用户名长度必须在4-10之间'})
  userName: string
  @IsNotEmpty({message: '密码不能为空'})
  @MinLength(6,{message: '密码长度必须大于6'})
  password: string
  @Min(18,{message: '年龄必须大于18岁'})
  @Max(100,{message: '年龄必须小于100岁'})
  age:number
  @IsNotEmpty({message: '性别不能为空'})
  gender: '男' | '女'
  @IsMobilePhone('zh-CN',{strictMode: false}, {message: '手机号码格式不符'})
  @Matches(/^1[34578]\d{9}$/,{message: '手机号码格式不符'})
  phone: string
}

const u = new User();
u.userName = 'lian'
u.password = '123456'
u.age = 20
u.gender = '男'
u.phone = '1373955422'
validate(u).then(errors => {
  log(errors)
})

    import axios from "axios";
import { log } from "console";
import 'reflect-metadata'
import{Type, plainToClass} from 'class-transformer'

class User {
  id: number;
  name:string
  @Type(() =>Number)
  age: number
  print() {
    console.log(this.id, this.name, this.age)
  }
}

  users.forEach((user:any) => {
    const u = plainToClass(User, user)

  })
})

    # 参数装饰器

    依赖注入，依赖倒置

    要求函数有三个参数：

    • 第一个参数：如果是静态方法，则为类本身，如果是实例方法，则为类的原型

    • 第二个参数：固定为字符串，表示方法名

    • 第三个参数：在参数列表中的索引

    import { log } from "console";

function test(target: any, propertyKey: string, index: number){
  log(target, propertyKey, index);
}

class MyMath {
  sum(a: number,@test b: number) {
    return a + b;
  }
}

    # TS自动注入元数据

    如果安装了reflect-metadata并且导入了该库，并且在成员上添加了元数据，并且*"emitDecoratorMetadata"*: true 启用，则TS在编译结果中，会将约束的类型，作为元数据加入到相应位置

    TS的类型检查（约束）将有机会在运行时进行

    import 'reflect-metadata'
class User {
  @Reflect.metadata('name', 'id')
  id: string;
}
