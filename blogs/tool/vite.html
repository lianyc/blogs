    vite基础

    11/9/2023 构建工具vite

    # 1.通过vite创建Vue3项目

    可自行选择使用框架以及使用语言

    npm init vite

    # 2.Vue3使用插件支持JSX

    yarn add @vitejs/plugin-vue-jsx

    vite.config.js

    import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import vueJsxPlugin from "@vitejs/plugin-vue-jsx";

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [vue(), vueJsxPlugin()],
})

    # 3.通过vite创建Vue2项目

    npm init vite // 使用vanilla创建项目
yarn add vite-plugin-vue2 -D
// vite.config.js
import { createVuePlugin } from 'vite-plugin-vue2'

export default {
  plugins: [
    createVuePlugin(/* options */)
  ],
}

    # 4.通过vite创建React项目

    npm init vite //选择react

    # 5.vite使用各种css框架

    vite已支持postcss,css modules

    • less

      yarn add less

    • sass

      yarn add sass

    # 6.vite中使用Typescript

    • isolatedModules:快速排出一些ts错误

      tsconfig.json:

      "compilerOptions": {
  "baseUrl": ".",
  "paths": {
    "@/*": ["./src/*"]
  },
  "isolatedModules":  true, // 这里
  "ignoreDeprecations": "5.0"
},

    • "types": ["vite/client"]：可以使用import.meta.env.VITE_BASE_API,import引入图片等

      "compilerOptions": {
  "baseUrl": ".",
  "paths": {
    "@/*": ["./src/*"]
  },
  "isolatedModules":  true,
  "ignoreDeprecations": "5.0",
  "types": ["vite/client"], // 这里
},

    # 7.vite中使用静态资源

    types:

    • url:

      import test from './test?url' //返回引入文件路径字符串

    • raw

      import test from './test?raw' //返回引入文件文本内容字符串

    • worker

      import Worker from './worker?worker'
const worker = new Worker()
worker.onmessage = function(e) {}

    json:

    import pkg from package.json
import {version} from package.json

    wasm:

    import init from 'fib.wasm'
init().then(m => {m.fib(10)})

    # 8.使用eslint和prettier

    yarn add eslint-plugin-import

yarn add eslint-plugin-node

yarn add eslint-pligin-promise

yarn add eslint-config-standard -D

    .eslintrc.js

    module.exports = {
	extends: "standard",
	rules: {
		"space-before-function-paren": "off" //方法名后需要空格关闭
	}
}

    .prettierrc

    {
  "semi": false,
  "singleQuote": true,
}

    scripts:

    "lint" : "eslint -- ext js src/"

    推送代码校验：

    yarn add husky -D
npx husky install
npx husky add .husky/pre commit "npm run lint"

    # 9.环境变量

    import.meta.env

    5个默认变量

    • MODE:模式，环境区分
    • BASE_URL: 初始URL
    • PROD: 正式环境
    • DEV:开发环境
    • SSR: 服务端渲染

    自定义变量：新建.env文件

    VITE_TITLE=HELLO

    .env.production: 正式环境生效

    .env.development：开发环境生效

    .env.test：测试环境生效

    .env.development.local：本地开发环境生效

    types:

    src/vite-env.d.ts

    interface ImportMetaEnv {
	VITE_TITLE: string
}
